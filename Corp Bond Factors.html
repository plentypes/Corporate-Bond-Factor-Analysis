Corporate Bond Factor Analysis

Data set: Dickerson, Alexander and Robotti, Cesare and Rossetti, Giulio, The Corporate Bond Factor Replication Crisis (January 18, 2026). Available at SSRN: https://ssrn.com/abstract

A 50-year monthly database of bond portfolios grouped by quantitative factors, used to evaluate fixed-income returns, turnover, and liquidity.

Initial Finding: High-yield systematic momentum factors (sysmom) demonstrate the strongest cumulative performance, with final values ranging from $8.09 to $21.65, significantly outperforming investment-grade strategies which cluster between $2.09 and $2.42.

    group_returns = clean_df[clean_df['factor'].isin(factor_list)]
    monthly_avg = group_returns.groupby('date')['return'].mean()

    ann_vol = monthly_avg.std() * np.sqrt(12)
    sharpe = ann_return / ann_vol

import pandas as pd
import numpy as np

costs = {
    'broad': 0.0010,    # 10 bps (IG Liquidity)
    'standard': 0.0025, # 25 bps (Mid-tier)
    'selective': 0.0040 # 40 bps (HY=Niche Liquidity)
}

def get_net_stats(factor_list, cost_rate, name):
    mask = (df['factor'].isin(factor_list)) & (df['leg'] == 'l') & (df['weighting'] == 'vw')
    group = df[mask].copy()
    
    monthly = group.groupby('date').agg({'return': 'mean', 'turnover': 'mean'}).sort_index()
    
    # Net Return = Gross - (Turnover * Cost)
    monthly['net_return'] = monthly['return'] - (monthly['turnover'] * cost_rate)
    
    final_wealth = (1 + monthly['net_return'].fillna(0)).cumprod().iloc[-1]
    ann_return = monthly['net_return'].mean() * 12
    ann_vol = monthly['net_return'].std() * np.sqrt(12)
    sharpe = ann_return / ann_vol if ann_vol != 0 else 0
    
    return {
        'Group': name,
        'Final Net Value ($)': round(final_wealth, 2),
        'Ann. Net Return': f"{ann_return:.2%}",
        'Ann. Net Vol': f"{ann_vol:.2%}",
        'Net Sharpe': round(sharpe, 3)
    }

    for factor in config['factors']:
        f_subset = df[(df['factor'] == factor) & (df['leg'] == 'l') & (df['weighting'] == 'vw')].copy()
        
        if f_subset.empty: continue
            
        r_type = 'exc' if 'exc' in f_subset['return_type'].unique() else f_subset['return_type'].unique()[0]
        f_data = f_subset[f_subset['return_type'] == r_type].sort_values('date')
        
        # Annualized Return: ((Final Value)^(1/years)) - 1
        n_years = len(f_data) / 12
        final_val = (1 + f_data['return'].fillna(0)).cumprod().iloc[-1]
        ann_return = (final_val ** (1/n_years)) - 1
        
        avg_turnover = f_data['turnover'].mean()
        avg_bonds = f_data['count'].mean()
        start_date = f_data['date'].min().strftime('%Y-%m-%d')
        
        # Sharpe
        ann_vol = f_data['return'].std() * np.sqrt(12)
        sharpe = (f_data['return'].mean() * 12) / ann_vol if ann_vol > 0 else 0
        
        f_data['net_return'] = f_data['return'] - (f_data['turnover'].fillna(0) * cost_rate)
        net_ann_vol = f_data['net_return'].std() * np.sqrt(12)
        net_sharpe = (f_data['net_return'].mean() * 12) / net_ann_vol if net_ann_vol > 0 else 0
        
        all_factor_results.append({
            "Group": group_name,
            "Factor": factor,
            "Ann. Return": f"{ann_return:.2%}",
            "Start": start_date,
            "Bonds": int(avg_bonds),
            "Sharpe": round(sharpe, 3),
            "Net Sharpe": round(net_sharpe, 3)
        })

summary_df = pd.DataFrame(all_factor_results)

print("SUMMARY FINDINGS")
print(summary_df.sort_values(['Group', 'Net Sharpe'], ascending=[True, False]).to_string(index=False))
